
# Make a new patch.
quilt new DH-5878

# Before modifying a file:
quilt add <filename>

# Makes quilt look for file system changes to be bundled into the patch.
quilt refresh

# Show the changes that have been made
quilt diff -z

# Take the patch off the filesystem
quilt pop

# List patches.
quilt applied
quilt unapplied

# Merge two patches together
# Apply the first patch 
quilt push first-patch-name
quilt series -v
# 'fold' in the second patch 
quilt fold < patches/second-patch-name
quilt refresh
# Delete the second patch from the series 
quilt delete second-patch-name
